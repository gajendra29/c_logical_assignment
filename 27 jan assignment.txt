void factorial(int a){
    int fact = 1;
    for(int i=1; i<=a; i++){
        fact = fact*i;
    }
    printf("The factorial is : %d\n",fact);
}
void fibo(int a){
    int n1 = 0;
    int n2 = 1;
    int n3;
    printf("The fibonacci is : %d %d",n1,n2);
    for(int i=1; i<=a-2; i++){
        n3=n1+n2;
        printf(" %d",n3);
        n1=n2;
        n2=n3;
    }
    printf("\n");
}
void reverse(int a){
    int sum = 0;
    int rem;
    while(a>0){
        rem = a%10;
        sum = sum*10+rem;
        a = a/10;
    }
    printf("The reverse is : %d\n",sum);
}
void pallindrom(int a){
    int sum = 0;
    int rem;
    int temp = a;
    while(a>0){
        rem = a%10;
        sum = sum*10+rem;
        a = a/10;
    }
    if(temp==sum){
        printf("It is pallindrom\n");
    }
    else{
        printf("Not a pallindrom\n");
    }
}

    b=a-b;
